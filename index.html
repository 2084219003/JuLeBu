3D坦克网页版 窗口同步器

发布者: Hu_Li 2025-08-01 3,567
游戏脚本

3D坦克窗口同步器，帮助刷机或多开玩家操作效率，智能识别窗口和名称。

// ==UserScript==
// @name         窗口同步器云仓库版
// @match        https://3dtank.com/play/*
// @grant        GM_xmlhttpRequest
// ==/UserScript==

(function () {
    'use strict';

    const SYNC_KEY = 'operation_sync';
    const WINDOW_ID_KEY = 'window_id';
    const CONTROLLED_WINDOWS_KEY = 'controlled_windows';
    const eventBlacklist = ['mousemove', 'mouseover', 'mouseout'];
    let isSyncing = false; // 默认停止同步
    let windowId;
    const broadcastChannel = new BroadcastChannel('window_sync_channel');
    const eventQueue = [];
    let lastOperation = null;
    const windowNames = {}; // 存储每个窗口的名称
    let throttleTimer = null;
    const THROTTLE_DELAY = 100; // 节流延迟时间，可根据实际情况调整
    let isControlled = false; // 标记当前窗口是否为被控窗口

    // 检查是否是首次打开页面
    if (!sessionStorage.getItem('first_load')) {
        localStorage.removeItem('all_window_ids');
        sessionStorage.setItem('first_load', 'true');
    }

    // 获取当前所有活动窗口 ID
    let allWindowIds = JSON.parse(localStorage.getItem('all_window_ids') || '[]');

    // 清理无效（不存在）的窗口 ID
    function cleanUpWindowIds() {
        allWindowIds = allWindowIds.filter(id => id);
        localStorage.setItem('all_window_ids', JSON.stringify(allWindowIds));
    }

    cleanUpWindowIds();

    // 为窗口分配唯一 ID
    if (!allWindowIds.length) {
        windowId = 1;
    } else {
        let newId = Math.max(...allWindowIds) + 1;
        windowId = newId;
    }
    allWindowIds.push(windowId);
    localStorage.setItem('all_window_ids', JSON.stringify(allWindowIds));
    localStorage.setItem(WINDOW_ID_KEY, windowId);

    // 初始窗口名称为等待加载
    let windowName = '等待名称加载';
    document.title = `${windowId}|${windowName}`;

    // 监听 DOM 变化，获取窗口名称
    const targetNode = document.body;
    const config = { childList: true, subtree: true };
    const callback = function (mutationsList) {
        for (const mutation of mutationsList) {
            const targetElement = document.querySelector(".UserInfoContainerStyle-containerProgressMainScreen > div:nth-child(1) > span");
            if (targetElement) {
                const newWindowName = targetElement.innerHTML;
                if (newWindowName) {
                    windowName = newWindowName;
                    document.title = `${windowId}|${windowName}`;
                    windowNames[windowId] = windowName; // 存储当前窗口名称
                    console.log(`当前窗口 ${windowId} 名称更新为: ${windowName}`);
                    try {
                        // 广播窗口名称
                        broadcastChannel.postMessage({ type: 'window_name', windowId: windowId, windowName: windowName });
                    } catch (error) {
                        console.error('广播窗口名称时出错:', error);
                    }
                    updateWindowSelection();
                    observer.disconnect();
                    break;
                }
            }
        }
    };
    const observer = new MutationObserver(callback);
    observer.observe(targetNode, config);

    // 页面加载完成后，广播新窗口的 ID 和名称信息
    document.addEventListener('DOMContentLoaded', function () {
        // 广播新窗口的 ID
        broadcastChannel.postMessage({ type: 'new_window', windowId: windowId });
        // 请求所有窗口的名称
        broadcastChannel.postMessage({ type: 'request_window_names' });
        // 如果已经获取到窗口名称，直接广播
        if (windowName!== '等待名称加载') {
            broadcastChannel.postMessage({ type: 'window_name', windowId: windowId, windowName: windowName });
        }
    });

    // 窗口关闭时通知其他窗口移除该窗口 ID
    window.addEventListener('beforeunload', function () {
        broadcastChannel.postMessage({ type: 'window_closed', windowId: windowId });
        allWindowIds = allWindowIds.filter(id => id!== windowId);
        localStorage.setItem('all_window_ids', JSON.stringify(allWindowIds));
        delete windowNames[windowId]; // 删除当前窗口名称
        updateWindowSelection();
    });

    // 创建开始和停止同步的按钮
    const buttonContainer = document.createElement('div');
    buttonContainer.style.position = 'fixed';
    buttonContainer.style.top = '20px';
    buttonContainer.style.left = '20px';
    buttonContainer.style.display = 'flex';
    buttonContainer.style.flexDirection = 'column';
    buttonContainer.style.gap = '15px';
    buttonContainer.style.zIndex = 1000;
    buttonContainer.style.padding = '20px';
    buttonContainer.style.borderRadius = '0px';
    buttonContainer.style.background = 'rgba(255, 255, 255, 0.1)';
    buttonContainer.style.backdropFilter = 'blur(10px)';
    buttonContainer.style.boxShadow = '0 8px 30px rgba(0, 0, 0, 0.15)';

    const startButton = document.createElement('button');
    startButton.innerHTML = ' 开始同步';
    startButton.style.padding = '18px 35px';
    startButton.style.border = 'none';
    startButton.style.borderRadius = '3px';
    startButton.style.background = 'linear-gradient(135deg, #3498db, #2ecc71)';
    startButton.style.color = 'white';
    startButton.style.fontSize = '18px';
    startButton.style.fontWeight = '600';
    startButton.style.cursor = 'pointer';
    startButton.style.boxShadow = '0 6px 15px rgba(52, 152, 219, 0.4)';
    startButton.style.transition = 'all 0.3s cubic-bezier(.25,.8,.25,1)';
    startButton.addEventListener('mouseover', function () {
        startButton.style.transform = 'scale(1.05)';
        startButton.style.boxShadow = '0 10px 20px rgba(52, 152, 219, 0.6)';
    });
    startButton.addEventListener('mouseout', function () {
        startButton.style.transform = 'scale(1)';
        startButton.style.boxShadow = '0 6px 15px rgba(52, 152, 219, 0.4)';
    });
    startButton.addEventListener('click', function () {
        isSyncing = true;
        startButton.disabled = true;
        stopButton.disabled = false;
        // 开始同步时重置状态
        eventQueue.length = 0;
        lastOperation = null;
        clearTimeout(throttleTimer);
        throttleTimer = null;

        // 禁用所有选中的复选框
        const checkboxes = windowSelectionDiv.querySelectorAll('input[type="checkbox"]:checked');
        checkboxes.forEach((checkbox) => {
            checkbox.disabled = true;
        });

        // 广播开始同步消息，通知其他窗口隐藏菜单
        broadcastChannel.postMessage({ type: 'start_sync', sourceWindowId: windowId });
    });

    const stopButton = document.createElement('button');
    stopButton.innerHTML = ' 停止同步';
    stopButton.style.padding = '18px 35px';
    stopButton.style.border = 'none';
    stopButton.style.borderRadius = '3px';
    stopButton.style.background = 'linear-gradient(135deg, #e74c3c, #c0392b)';
    stopButton.style.color = 'white';
    stopButton.style.fontSize = '18px';
    stopButton.style.fontWeight = '600';
    stopButton.style.cursor = 'pointer';
    stopButton.style.boxShadow = '0 6px 15px rgba(231, 76, 60, 0.4)';
    stopButton.style.transition = 'all 0.3s cubic-bezier(.25,.8,.25,1)';
    stopButton.addEventListener('mouseover', function () {
        stopButton.style.transform = 'scale(1.05)';
        stopButton.style.boxShadow = '0 10px 20px rgba(231, 76, 60, 0.6)';
    });
    stopButton.addEventListener('mouseout', function () {
        stopButton.style.transform = 'scale(1)';
        stopButton.style.boxShadow = '0 6px 15px rgba(231, 76, 60, 0.4)';
    });
    stopButton.disabled = false;
    stopButton.addEventListener('click', function () {
        isSyncing = false;
        isControlled = false; // 停止同步时，将被控状态置为 false
        startButton.disabled = false;
        stopButton.disabled = true;
        // 停止同步时重置状态
        eventQueue.length = 0;
        lastOperation = null;
        clearTimeout(throttleTimer);
        throttleTimer = null;

        // 启用所有复选框
        const checkboxes = windowSelectionDiv.querySelectorAll('input[type="checkbox"]');
        checkboxes.forEach((checkbox) => {
            checkbox.disabled = false;
        });

        // 广播停止同步消息，通知其他窗口显示菜单
        broadcastChannel.postMessage({ type: 'stop_sync', sourceWindowId: windowId, isMaster: true });
    });

    buttonContainer.appendChild(startButton);
    buttonContainer.appendChild(stopButton);

    // 创建窗口选择界面
    const windowSelectionDiv = document.createElement('div');
    windowSelectionDiv.style.position = 'fixed';
    windowSelectionDiv.style.top = '210px';
    windowSelectionDiv.style.left = '20px';
    windowSelectionDiv.style.width = '300px';
    windowSelectionDiv.style.padding = '25px';
    windowSelectionDiv.style.borderRadius = '0px';
    windowSelectionDiv.style.background = 'rgba(255, 255, 255, 0.1)';
    windowSelectionDiv.style.backdropFilter = 'blur(10px)';
    windowSelectionDiv.style.boxShadow = '0 8px 30px rgba(0, 0, 0, 0.15)';
    windowSelectionDiv.style.zIndex = 1000;

    const label = document.createElement('label');
    label.textContent = '受控窗口: ';
    label.style.color = 'white';
    label.style.fontSize = '20px';
    label.style.fontWeight = '600';
    label.style.marginBottom = '15px';
    windowSelectionDiv.appendChild(label);

    function updateWindowSelection() {
    allWindowIds = JSON.parse(localStorage.getItem('all_window_ids') || '[]');
    const existingCheckboxes = windowSelectionDiv.querySelectorAll('input[type="checkbox"]');
    const existingIds = Array.from(existingCheckboxes).map(cb => Number(cb.value));
    allWindowIds.forEach((id) => {
        if (id!== windowId) {
            let checkbox = windowSelectionDiv.querySelector(`#window-${id}`);
            if (!checkbox) {
                checkbox = document.createElement('input');
                checkbox.type = 'checkbox';
                checkbox.id = `window-${id}`;
                checkbox.value = id;
                checkbox.style.appearance = 'none';
                checkbox.style.width = '25px';
                checkbox.style.height = '25px';
                checkbox.style.border = '2px solid white';
                checkbox.style.borderRadius = '5px';
                checkbox.style.cursor = 'pointer';
                checkbox.style.transition = 'all 0.3s ease';

                const checkboxLabel = document.createElement('label');
                const otherWindowName = windowNames[id] || '等待名称加载';
                checkboxLabel.textContent = otherWindowName;
                checkboxLabel.htmlFor = `window-${id}`;
                checkboxLabel.style.color = 'white';
                checkboxLabel.style.fontSize = '18px';
                checkboxLabel.style.marginLeft = '15px';

                const checkboxContainer = document.createElement('div');
                checkboxContainer.style.display = 'flex';
                checkboxContainer.style.alignItems = 'center';
                checkboxContainer.style.marginBottom = '20px';
                checkboxContainer.appendChild(checkbox);
                checkboxContainer.appendChild(checkboxLabel);

                windowSelectionDiv.appendChild(checkboxContainer);

                checkbox.addEventListener('change', function () {
                    const imgUrl = 'https://tankionline.com/favicon.ico';
                    if (this.checked) {
                        this.style.backgroundImage = `url(${imgUrl})`;
                        this.style.backgroundSize = 'cover';
                        this.style.borderColor = 'transparent';
                        // 新增这一行，避免背景图片重复
                        this.style.backgroundRepeat = 'no-repeat';
                    } else {
                        this.style.backgroundImage = 'none';
                        this.style.borderColor = 'white';
                    }
                    // 选中状态改变时保存受控窗口设置
                    const checkboxes = windowSelectionDiv.querySelectorAll('input[type="checkbox"]');
                    const controlledWindows = [];
                    checkboxes.forEach((cb) => {
                        if (cb.checked) {
                            controlledWindows.push(Number(cb.value));
                        }
                    });
                    localStorage.setItem(CONTROLLED_WINDOWS_KEY, JSON.stringify(controlledWindows));
                });
            } else {
                const checkboxLabel = checkbox.nextElementSibling;
                const otherWindowName = windowNames[id] || '等待名称加载';
                checkboxLabel.textContent = otherWindowName;
            }
        }
    });
    // 移除已关闭窗口的 checkbox
    existingIds.forEach((id) => {
        if (!allWindowIds.includes(id)) {
            const checkbox = windowSelectionDiv.querySelector(`#window-${id}`);
            if (checkbox) {
                const container = checkbox.parentNode;
                windowSelectionDiv.removeChild(container);
            }
        }
    });
    console.log('窗口选择界面已更新');
}

    updateWindowSelection();

    document.body.appendChild(buttonContainer);
    document.body.appendChild(windowSelectionDiv);

    // 引入Font Awesome图标库（需联网加载）
    const link = document.createElement('link');
    link.rel ='stylesheet';
    link.href = 'https://cdnjs.cloudflare.com/ajax/libs/font-awesome/5.15.4/css/all.min.css';
    document.head.appendChild(link);

    // 监听 storage 变化，重放操作
    window.addEventListener('storage', function (event) {
        if (event.key === SYNC_KEY) {
            try {
                const operation = JSON.parse(event.newValue);
                if (shouldReceiveOperation(operation) &&!isDuplicateOperation(operation)) {
                    console.log('接收到 storage 操作消息:', operation);
                    eventQueue.push(operation);
                    lastOperation = operation;
                    processEventQueue();
                }
            } catch (error) {
                console.error('解析操作时出错:', error);
            }
        }
    });

    // 监听广播消息
    broadcastChannel.addEventListener('message', function (event) {
        const data = event.data;
        console.log('接收到广播消息:', data);
        if (data.type === 'new_window') {
            const newWindowId = data.windowId;
            if (!allWindowIds.includes(newWindowId)) {
                allWindowIds.push(newWindowId);
                localStorage.setItem('all_window_ids', JSON.stringify(allWindowIds));
                updateWindowSelection();
            }
        } else if (data.type === 'window_closed') {
            const closedWindowId = data.windowId;
            allWindowIds = allWindowIds.filter(id => id!== closedWindowId);
            localStorage.setItem('all_window_ids', JSON.stringify(allWindowIds));
            delete windowNames[closedWindowId]; // 删除关闭窗口的名称
            updateWindowSelection();
        } else if (data.type === 'window_name') {
            console.log(`接收到窗口 ${data.windowId} 名称更新为: ${data.windowName}`);
windowNames[data.windowId] = data.windowName; // 更新窗口名称
updateWindowSelection();
} else if (data.type ==='request_window_names') {
    if (windowName!== '等待名称加载') {
        broadcastChannel.postMessage({ type: 'window_name', windowId: windowId, windowName: windowName });
    }
} else if (data.type ==='start_sync') {
    const controlledWindows = JSON.parse(localStorage.getItem(CONTROLLED_WINDOWS_KEY) || '[]');
    console.log(`当前窗口 ID: ${windowId}, 消息源窗口 ID: ${data.sourceWindowId}, 受控窗口列表: ${controlledWindows}`);
    if (data.sourceWindowId!== windowId && controlledWindows.includes(windowId)) {
        console.log('隐藏菜单，当前窗口为被控窗口');
        isControlled = true;
        startButton.disabled = true;
        startButton.innerHTML = '正在被控';
        if (windowSelectionDiv.style.display!== 'none') {
            windowSelectionDiv.style.display = 'none';
            console.log('已将菜单显示状态设置为 none');
        }
    }
} else if (data.type ==='stop_sync') {
    const controlledWindows = JSON.parse(localStorage.getItem(CONTROLLED_WINDOWS_KEY) || '[]');
    console.log(`当前窗口 ID: ${windowId}, 消息源窗口 ID: ${data.sourceWindowId}, 受控窗口列表: ${controlledWindows}`);
    if (data.isMaster && data.sourceWindowId!== windowId && controlledWindows.includes(windowId)) {
        console.log('显示菜单，当前窗口为被控窗口，且是主控窗口发起的停止同步');
        isControlled = false;
        startButton.disabled = false;
        startButton.innerHTML = ' 开始同步';
        if (windowSelectionDiv.style.display === 'none') {
            windowSelectionDiv.style.display = 'block';
            console.log('已将菜单显示状态设置为 block');
        }
    } else if (!data.isMaster && windowId === data.sourceWindowId) {
        // 如果是被控窗口自己发起的停止同步，只重置自身状态，不更新其他窗口按钮文本

        isControlled = false;
        startButton.disabled = false;
        startButton.innerHTML = ' 开始同步';

        eventQueue.length = 0;
        lastOperation = null;
        clearTimeout(throttleTimer);
        throttleTimer = null;
        const checkboxes = windowSelectionDiv.querySelectorAll('input[type="checkbox"]');
        checkboxes.forEach((checkbox) => {
            checkbox.disabled = false;
        });
    }
} else {
    const operation = data;
    if (shouldReceiveOperation(operation) &&!isDuplicateOperation(operation)) {
        console.log('接收到广播操作消息:', operation);
        eventQueue.push(operation);
        lastOperation = operation;
        processEventQueue();
    }
}
});

// 捕获操作并存储到 localStorage

function captureAndSyncOperation(event) {
    if (eventBlacklist.includes(event.type)) {
        return;
    }
    // 暂时移除节流逻辑
    // if (throttleTimer) {
    //     return;
    // }
    // throttleTimer = setTimeout(() => {
    //     throttleTimer = null;
    // }, THROTTLE_DELAY);

    const target = event.target;
    const operation = {
        eventType: event.type,
        sourceWindowId: windowId
    };
    if (event.type === 'click') {
        // 记录绝对坐标
        operation.clientX = event.clientX;
        operation.clientY = event.clientY;
    } else if (event.type === 'input') {
        const selector = getUniqueSelector(target);
        operation.selector = selector;
        operation.value = target.value;
    } else if (event.type === 'keydown' || event.type === 'keyup') {
        operation.key = event.key;
        operation.code = event.code;
        operation.which = event.which;
        operation.keyCode = event.keyCode;
    }
    try {
        localStorage.setItem(SYNC_KEY, JSON.stringify(operation));
        broadcastChannel.postMessage(operation);
    } catch (error) {
        console.error('捕获并同步操作时出错:', error);
    }
}

function replayOperation(operation) {
    if (operation.eventType === 'click') {
        const clientX = operation.clientX;
        const clientY = operation.clientY;

        // 检查 window 对象是否具有必要的属性和方法
        if (typeof window!== 'undefined' && window.document && typeof window.document.elementFromPoint === 'function') {
            console.log('当前 window 对象有效，继续创建鼠标事件');
            try {
                const view = window.document.defaultView;
                // 创建并触发 mousedown 事件
                const mouseDownEvent = new MouseEvent('mousedown', {
                    view: view,
                    bubbles: true,
                    cancelable: true,
                    clientX: clientX,
                    clientY: clientY
                });
                const targetElement = document.elementFromPoint(clientX, clientY);
                if (targetElement) {
                    targetElement.dispatchEvent(mouseDownEvent);
                }

                // 创建并触发 mouseup 事件
                const mouseUpEvent = new MouseEvent('mouseup', {
                    view: view,
                    bubbles: true,
                    cancelable: true,
                    clientX: clientX,
                    clientY: clientY
                });
                if (targetElement) {
                    targetElement.dispatchEvent(mouseUpEvent);
                }

                // 创建并触发 click 事件
                const clickEvent = new MouseEvent('click', {
                    view: view,
                    bubbles: true,
                    cancelable: true,
                    clientX: clientX,
                    clientY: clientY
                });
                if (targetElement) {
                    targetElement.dispatchEvent(clickEvent);
                }
            } catch (error) {
                console.error('创建鼠标事件时出错:', error);
                console.log('window 对象信息:', window);
            }
        } else {
            console.error('window 对象无效，无法创建鼠标事件');
            console.log('window 对象信息:', window);
        }
    } else if (operation.eventType === 'input') {
        const targetElement = document.querySelector(operation.selector);
        if (targetElement) {
            targetElement.focus();
            targetElement.value = operation.value;
            const inputEvent = new Event('input', {
                bubbles: true,
                cancelable: true
            });
            targetElement.dispatchEvent(inputEvent);
        }
    } else if (operation.eventType === 'keydown' || operation.eventType === 'keyup') {
        const targetElement = document.activeElement;
        if (targetElement) {
            targetElement.focus();
            const keyEvent = new KeyboardEvent(operation.eventType, {
                key: operation.key,
                code: operation.code,
                which: operation.which,
                keyCode: operation.keyCode,
                bubbles: true,
                cancelable: true
            });
            targetElement.dispatchEvent(keyEvent);
        }
    }
}

function shouldReceiveOperation(operation) {
    const controlledWindows = JSON.parse(localStorage.getItem(CONTROLLED_WINDOWS_KEY) || '[]');
    return controlledWindows.includes(windowId) && operation.sourceWindowId!== windowId;
}

function isDuplicateOperation(operation) {
    if (!lastOperation) return false;
    if (operation.eventType === 'keydown' || operation.eventType === 'keyup') {
        return operation.eventType === lastOperation.eventType && operation.key === lastOperation.key;
    }
    return JSON.stringify(operation) === JSON.stringify(lastOperation);
}

function getUniqueSelector(element) {
    const path = [];
    while (element && element.nodeType === Node.ELEMENT_NODE) {
        let selector = element.nodeName.toLowerCase();
        if (element.id) {
            selector += `#${element.id}`;
        } else if (element.className) {
            const classNames = element.className.trim().split(/\s+/);
            classNames.forEach((className) => {
                selector += `.${className}`;
            });
        }
        // 添加元素在父节点中的索引信息
        const siblings = Array.from(element.parentNode.children);
        const index = siblings.indexOf(element);
        if (index > 0) {
            selector += `:nth-child(${index + 1})`;
        }
        path.unshift(selector);
        element = element.parentNode;
    }
    return path.join(' > ');
}

function processEventQueue() {
    if (eventQueue.length > 0) {
        const operation = eventQueue.shift();
        replayOperation(operation);
        if (eventQueue.length > 0) {
            processEventQueue();
        }
    }
}
